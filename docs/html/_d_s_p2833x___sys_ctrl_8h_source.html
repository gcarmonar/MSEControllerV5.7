187  Uint16 SCSR; // 18: System control and status register
188  Uint16 WDCNTR; // 19: WD counter register
189  Uint16 rsvd4; // 20
190  Uint16 WDKEY; // 21: WD reset key register
191  Uint16 rsvd5[3]; // 22-24
