68  // 0 only interrupt event triggers single burst transfer

71  // 0 stop when transfer count is zero
72  // 1 re-initialize when transfer count is zero
73  Uint16 SYNCE:1; // 12 Sync Enable Bit (R/W):
74  // 0 ignore selected interrupt sync signal
75  // 1 enable selected interrupt sync signal
76  Uint16 SYNCSEL:1; // 13 Sync Select Bit (R/W):
77  // 0 sync signal controls source wrap counter
78  // 1 sync signal controls destination wrap counter
